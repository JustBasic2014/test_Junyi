def reverse(string):
    temp = list(string)

    left = 0
    right = len(temp)-1

    while left <= right:
        temp[left], temp[right] = temp[right], temp[left]
        left += 1
        right -= 1

    result = ''.join(temp)

    return result

def reverse_sentence(sentence):
    if sentence is None or len(sentence) == 0:
        return ""
        
    temp_list = sentence.split(" ")

    for i in range(len(temp_list)):
        temp_list[i] = reverse(temp_list[i])

    result = ' '.join(temp_list)

    return result

def common_multiple(num):
    count = 0

    for i in range(num):
        if i % 3 == 0 and i % 5 == 0:
            count += 1
        elif i % 3 == 0:
            continue
        elif i % 5 == 0:
            continue
        else:
            count += 1

    return count
